32def best_effort_rollback(con: duckdb.DuckDBPyConnection, *, context: str) -> None: 

33 """Issue ROLLBACK without letting a failed rollback mask the original error. 

34 

35 Use in `except Exception: best_effort_rollback(con, context=...); raise` 

36 blocks. If ROLLBACK itself raises (broken connection, txn already aborted 

37 by an earlier error inside the same statement, etc.) we log the 

38 secondary failure and swallow it so the caller's `raise` re-raises the 

39 *original* exception. The original cause is what an operator needs in 

40 the traceback; a "ROLLBACK failed because connection is closed" 

41 secondary error just buries the real bug. 

42 """ 

43 try: 

44 con.execute("ROLLBACK") 

45 except Exception: 

46 logger.exception("Best-effort ROLLBACK failed (context: %s)", context)

153def get_budget_connection(year: int) -> duckdb.DuckDBPyConnection: 

154 """Open a connection to budget_YYYY.duckdb, creating it if needed. 

155 

156 The caller is responsible for closing the connection. 

157 The connection has config.duckdb ATTACHed as 'config' (READ_ONLY). 

158 """ 

159 path = init_budget_db(year) 

160 con = duckdb.connect(str(path)) 

161 try: 

162 # DuckDB's ATTACH does not accept the path as a bound parameter, so 

163 # we have to interpolate. SQL-escape the single quote so a path 

164 # containing `'` (perfectly legal on macOS user dirs and in tmp 

165 # paths used by tests) cannot break out of the string literal. 

166 config_path_sql = str(CONFIG_DB).replace("'", "''") 

167 con.execute(f"ATTACH '{config_path_sql}' AS config (READ_ONLY)") 

168 except Exception: 

169 con.close() 

170 raise 

171 return con

284def forward_projection( 

285 con: duckdb.DuckDBPyConnection, 

286 *, 

287 latest_sheet_year: int, 

288 category_id: int, 

289 event_id: int | None, 

290 tag_ids: list[int] | set[int] | tuple[int, ...], 

291) -> tuple[str, str] | None: 

292 """Resolve `(category_id, event_id, tag set)` to `(sheet_category, sheet_group)`. 

293 

294 Lookup order against `sheet_mapping` + `sheet_mapping_tags` inside 

295 `latest_sheet_year`: 

296 1. exact match: same `category_id`, same `event_id` (NULL matches NULL), 

297 and same tag-id set as the expense; 

298 2. fallback: first `sheet_mapping` row with the same `category_id`, 

299 ignoring `event_id` and tags. 

300 Ties inside the same preference bucket are broken by `sheet_mapping.id ASC`. 

301 """ 

302 expected_tags = sorted(int(t) for t in tag_ids) 

303 candidates = fetchall_as( 

304 ForwardProjectionCandidateRow, 

305 con, 

306 load_sql("forward_projection.sql"), 

307 [latest_sheet_year, category_id], 

308 ) 

309 if not candidates: 

310 return None 

311 

312 for cand in candidates: 

313 cand_tags = sorted(int(t) for t in cand.tag_ids) 

314 if cand.event_id == event_id and cand_tags == expected_tags: 

315 return (cand.sheet_category, cand.sheet_group) 

316 

317 first = candidates[0] 

318 return (first.sheet_category, first.sheet_group)

326def reserve_expense_id_year(expense_id: str, year: int) -> tuple[int, bool]: 

327 """Atomically claim or look up the year that owns `expense_id`. 

328 

329 Returns `(stored_year, newly_inserted)`: 

330 * `stored_year` is the year currently associated with the id after the 

331 call (may differ from `year` if the row already existed — the API 

332 layer turns that into a 409 cross-year reuse error). 

333 * `newly_inserted` is True iff this call performed the INSERT. The 

334 caller MUST `release_expense_id_year(expense_id)` if a downstream 

335 step fails, so the registry doesn't end up holding a phantom row 

336 for an expense that never made it into the budget DB. 

337 

338 Concurrency: DuckDB doesn't give us a true row-level "SELECT FOR UPDATE", 

339 so two concurrent callers can both fall through the `SELECT NULL` branch 

340 and try to INSERT. The PK constraint on `expense_id` makes the loser's 

341 INSERT fail with `duckdb.ConstraintException`; we catch that here and 

342 re-read so the loser sees the same `(stored_year, False)` result it 

343 would have gotten if it had simply run a fraction of a second later. 

344 Without this, the loser would surface a 5xx instead of a clean 409. 

345 """ 

346 con = get_config_connection(read_only=False) 

347 try: 

348 # `txn_active` lets the outer except skip its own rollback when 

349 # the inner ConstraintException path has already issued one. The 

350 # alternative — a second `best_effort_rollback` against an 

351 # already-closed txn — produces a misleading "ROLLBACK failed" 

352 # log that masks the real RuntimeError("vanished row") path. 

353 txn_active = True 

354 con.execute("BEGIN") 

355 try: 

356 row = con.execute( 

357 "SELECT year FROM expense_id_registry WHERE expense_id = ?", 

358 [expense_id], 

359 ).fetchone() 

360 if row is not None: 

361 con.execute("COMMIT") 

362 txn_active = False 

363 return int(row[0]), False 

364 try: 

365 con.execute( 

366 "INSERT INTO expense_id_registry (expense_id, year) VALUES (?, ?)", 

367 [expense_id, year], 

368 ) 

369 except duckdb.ConstraintException: 

370 # Lost a race to a concurrent reserver; the row is now there. 

371 # Rollback the failed INSERT (ROLLBACK errors here would 

372 # mean the connection is unusable anyway, so we let them 

373 # propagate — unlike the outer except block, this path 

374 # MUST roll back to release the txn before the SELECT 

375 # below can succeed). 

376 con.execute("ROLLBACK") 

377 txn_active = False 

378 raced = con.execute( 

379 "SELECT year FROM expense_id_registry WHERE expense_id = ?", 

380 [expense_id], 

381 ).fetchone() 

382 if raced is None: 

383 # Constraint fired but row vanished — should be impossible 

384 # for a PK insert, but bubble up as-is rather than guess. 

385 msg = ( 

386 f"expense_id_registry race for {expense_id!r}: PK " 

387 "violation on INSERT but row not present on re-read" 

388 ) 

389 raise RuntimeError(msg) from None 

390 return int(raced[0]), False 

391 con.execute("COMMIT") 

392 txn_active = False 

393 return year, True 

394 except Exception: 

395 if txn_active: 

396 best_effort_rollback( 

397 con, 

398 context=f"reserve_expense_id({expense_id!r})", 

399 ) 

400 raise 

401 finally: 

402 con.close()

435def insert_expense( # noqa: PLR0913, C901 

436 con: duckdb.DuckDBPyConnection, 

437 *, 

438 expense_id: str, 

439 expense_datetime: datetime, 

440 amount: float, 

441 amount_original: float, 

442 currency_original: str, 

443 category_id: int, 

444 event_id: int | None = None, 

445 comment: str = "", 

446 sheet_category: str | None = None, 

447 sheet_group: str | None = None, 

448 tag_ids: list[int] | None = None, 

449 enqueue_sync: bool = True, 

450) -> InsertExpenseResult: 

451 """Insert an expense + tags + queue row in one transaction. 

452 

453 Returns 'created', 'duplicate', or 'conflict'. 

454 Validates `category_id` against ATTACHed config.categories and `event_id` 

455 against ATTACHed config.events. Tag IDs are validated against 

456 config.tags. Provenance pair `(sheet_category, sheet_group)` must be 

457 NULL/NULL or both non-NULL — this is a runtime invariant enforced here 

458 instead of a CHECK so the migration stays simple. 

459 """ 

460 if (sheet_category is None) != (sheet_group is None): 

461 msg = ( 

462 "sheet_category and sheet_group must be both NULL (runtime row) " 

463 "or both non-NULL (bootstrap-imported provenance row)" 

464 ) 

465 raise ValueError(msg) 

466 

467 tag_ids = tag_ids or [] 

468 

469 con.execute("BEGIN") 

470 try: 

471 if not con.execute( 

472 "SELECT 1 FROM config.categories WHERE id = ?", 

473 [category_id], 

474 ).fetchone(): 

475 raise ValueError(f"category_id {category_id} not found in config.categories") 

476 if ( 

477 event_id is not None 

478 and not con.execute( 

479 "SELECT 1 FROM config.events WHERE id = ?", 

480 [event_id], 

481 ).fetchone() 

482 ): 

483 raise ValueError(f"event_id {event_id} not found in config.events") 

484 for tid in tag_ids: 

485 if not con.execute( 

486 "SELECT 1 FROM config.tags WHERE id = ?", 

487 [tid], 

488 ).fetchone(): 

489 raise ValueError(f"tag_id {tid} not found in config.tags") 

490 

491 inserted = con.execute( 

492 load_sql("insert_expense.sql"), 

493 [ 

494 expense_id, 

495 expense_datetime, 

496 amount, 

497 amount_original, 

498 currency_original, 

499 category_id, 

500 event_id, 

501 comment, 

502 sheet_category, 

503 sheet_group, 

504 ], 

505 ).fetchone() 

506 

507 if inserted is not None: 

508 for tid in tag_ids: 

509 con.execute( 

510 "INSERT INTO expense_tags (expense_id, tag_id) VALUES (?, ?)" 

511 " ON CONFLICT DO NOTHING", 

512 [expense_id, tid], 

513 ) 

514 if enqueue_sync: 

515 con.execute( 

516 "INSERT INTO sheet_sync_jobs (expense_id, status) VALUES (?, 'pending')" 

517 " ON CONFLICT DO NOTHING", 

518 [expense_id], 

519 ) 

520 con.execute("COMMIT") 

521 return "created" 

522 

523 existing = fetchone_as( 

524 ExistingExpenseRow, 

525 con, 

526 load_sql("get_existing_expense.sql"), 

527 [expense_id], 

528 ) 

529 assert existing is not None, f"expense {expense_id} vanished after ON CONFLICT" 

530 

531 existing_tag_ids = sorted( 

532 int(r[0]) 

533 for r in con.execute( 

534 "SELECT tag_id FROM expense_tags WHERE expense_id = ?", 

535 [expense_id], 

536 ).fetchall() 

537 ) 

538 incoming_tag_ids = sorted(int(t) for t in tag_ids) 

539 

540 stored = ( 

541 existing.amount, 

542 existing.amount_original, 

543 existing.currency_original, 

544 existing.category_id, 

545 existing.event_id, 

546 existing.comment, 

547 existing.datetime, 

548 existing.sheet_category, 

549 existing.sheet_group, 

550 existing_tag_ids, 

551 ) 

552 incoming = ( 

553 Decimal(str(amount)), 

554 Decimal(str(amount_original)), 

555 currency_original, 

556 category_id, 

557 event_id, 

558 comment, 

559 expense_datetime, 

560 sheet_category, 

561 sheet_group, 

562 incoming_tag_ids, 

563 ) 

564 

565 # NOT best-effort: this is the SUCCESS path (no INSERT 

566 # happened, "duplicate"/"conflict" doesn't write). A failing 

567 # ROLLBACK here is the primary error — bubbling it out is 

568 # correct, masking it would leave the txn open. 

569 con.execute("ROLLBACK") 

570 

571 if stored == incoming: 

572 return "duplicate" 

573 return "conflict" 

574 

575 except Exception: 

576 # The outer except catches genuine failures (the SQL above 

577 # raised). Use best-effort rollback so a secondary ROLLBACK 

578 # failure doesn't replace the original error in the traceback. 

579 best_effort_rollback(con, context=f"insert_expense({expense_id!r})") 

580 raise

622def claim_sync_job( 

623 con: duckdb.DuckDBPyConnection, 

624 expense_id: str, 

625 *, 

626 claim_token: str | None = None, 

627 now: datetime | None = None, 

628 stale_before: datetime | None = None, 

629) -> str | None: 

630 """Atomically claim `expense_id`. Returns the claim_token on success, None otherwise. 

631 

632 A row is claimable if it is `pending`, OR if it is `in_progress` with 

633 `claimed_at < stale_before` (lease-style stale-claim recovery). 

634 """ 

635 if claim_token is None: 

636 claim_token = uuid.uuid4().hex 

637 if now is None: 

638 now = datetime.now() 

639 if stale_before is None: 

640 stale_before = now - DEFAULT_CLAIM_STALE_TIMEOUT 

641 

642 con.execute("BEGIN") 

643 try: 

644 row = con.execute( 

645 "SELECT status, claim_token, claimed_at FROM sheet_sync_jobs WHERE expense_id = ?", 

646 [expense_id], 

647 ).fetchone() 

648 if row is None: 

649 con.execute("COMMIT") 

650 return None 

651 status, _existing_token, claimed_at = row 

652 

653 is_pending = status == "pending" 

654 is_stale = status == "in_progress" and claimed_at is not None and claimed_at < stale_before 

655 if not (is_pending or is_stale): 

656 con.execute("COMMIT") 

657 return None 

658 

659 con.execute( 

660 "UPDATE sheet_sync_jobs SET status = 'in_progress'," 

661 " claim_token = ?, claimed_at = ? WHERE expense_id = ?", 

662 [claim_token, now, expense_id], 

663 ) 

664 con.execute("COMMIT") 

665 return claim_token 

666 except duckdb.TransactionException: 

667 # DuckDB uses optimistic concurrency control; under concurrent 

668 # claim attempts on the same row the loser gets a transaction 

669 # conflict here. Treat that as "not claimable right now" — the 

670 # winner already moved the row to `in_progress`, so the next 

671 # sweep (or the in-flight worker) will own it. Returning None 

672 # avoids blowing up the sweep with a noisy exception that the 

673 # caller would have to translate back to "skip this row" anyway. 

674 best_effort_rollback(con, context=f"claim_sync_job({expense_id}) txn conflict") 

675 logger.debug("Transaction conflict claiming %s; another worker won", expense_id) 

676 return None 

677 except Exception: 

678 # Best-effort rollback: if ROLLBACK itself raises (broken 

679 # connection, constraint surfacing only at boundary, etc.) we 

680 # log and re-raise the *original* exception, not the secondary 

681 # one, so the caller's traceback points at the real cause 

682 # instead of a confusing rollback failure that masks it. 

683 best_effort_rollback(con, context=f"claim_sync_job({expense_id}) generic error") 

684 raise

702 return len(rows) > 0

705def _delete_sync_job( 

706 con: duckdb.DuckDBPyConnection, 

707 expense_id: str, 

708 *, 

709 claim_token: str | None, 

710) -> bool: 

711 """Single source of truth for deleting a `sheet_sync_jobs` row. 

712 

713 `claim_token=None` skips the token check (force-delete by expense_id 

714 only). Factored out so a future column change to `sheet_sync_jobs` 

715 only updates one DELETE statement instead of two divergent ones; the 

716 public `clear_sync_job` and `force_clear_sync_job` are thin wrappers 

717 that document the two distinct legitimate use cases. 

718 """ 

719 if claim_token is None: 

720 rows = con.execute( 

721 "DELETE FROM sheet_sync_jobs WHERE expense_id = ? RETURNING expense_id", 

722 [expense_id], 

723 ).fetchall() 

724 else: 

725 rows = con.execute( 

726 "DELETE FROM sheet_sync_jobs WHERE expense_id = ? AND claim_token = ?" 

727 " RETURNING expense_id", 

728 [expense_id, claim_token], 

729 ).fetchall() 

730 return len(rows) > 0

742def force_clear_sync_job(con: duckdb.DuckDBPyConnection, expense_id: str) -> bool: 

743 """Delete the queue row by expense_id, ignoring claim_token. 

744 

745 Use ONLY when the caller has definitively succeeded at the side-effect 

746 that the queue row was tracking (e.g. the Sheets append already 

747 happened) and a normal claim-token-protected `clear_sync_job` failed 

748 because the claim was stolen by a stale-claim sweep. Deleting 

749 unconditionally here prevents a *third* sheet append: the thief who 

750 stole the claim will (or already did) cause the second append, and 

751 leaving the row in place would cause the next sweep to claim and 

752 append a third copy. 

753 

754 Returns True if a row was deleted. 

755 """ 

756 return _delete_sync_job(con, expense_id, claim_token=None)
