1"""Runtime classification taxonomy for ``dinary.db``. 

2 

3This file is the authoritative source of truth for the runtime catalog 

4tables: ``category_groups``, ``categories``, ``tags``, and ``events``. 

5When this file disagrees with ``docs/src/ru/taxonomy.md``, the code 

6wins. 

7 

8Phase 1 PWA hardcodes the same tag dictionary defined here. There is 

9no ``/api/tags`` endpoint. 

10 

11Scope of this module 

12-------------------- 

13 

14Only the *runtime* catalog lives here — the hardcoded 3D vocabulary 

15(``ENTRY_GROUPS``, ``PHASE1_TAGS``, event constants, ``EXPLICIT_EVENTS``) 

16plus the FK-safe in-place upsert primitives and the single public 

17entry point ``seed_classification_catalog``. The ``import_mapping`` 

18table, the legacy 2D→3D derivation rules, and the 

19``.deploy/import_sources.json``-driven category discovery all live 

20in ``dinary.imports.seed``. The direction of dependency is strictly 

21``imports.seed -> services.seed_config`` (never the reverse); runtime 

22code never reaches into ``imports.seed``, and non-import deployments 

23never import it. 

24 

25Non-import bootstrap: ``bootstrap_catalog()`` provides a zero-Sheets 

26entry point that seeds only the hardcoded runtime taxonomy. This is 

27what ``inv bootstrap-catalog`` invokes on every deploy, so every 

28deployment lands with a populated catalog even when the operator has 

29no historical Google Sheets to import from. 

30 

31FK-safe in-place sync 

32--------------------- 

33 

34``seed_classification_catalog`` never deletes the DB file and never 

35renumbers catalog ids. Ledger tables (``expenses``, ``expense_tags``, 

36``sheet_logging_jobs``, ``income``) carry real FKs into the catalog; 

37deleting and renumbering referenced rows would violate those FKs. 

38 

39The sync algorithm: 

40 

411. Mark every catalog row ``is_active=FALSE``. Rows that reappear in 

42 the new taxonomy snapshot get flipped back to ``TRUE``; rows that 

43 don't stay ``FALSE`` and are hidden from the live API while 

44 remaining FK-valid targets for historical ledger rows. 

452. Upsert groups / categories / tags / events by ``name`` (the 

46 natural key). Pre-existing rows keep their integer ``id``; new 

47 rows get a fresh ``id = max(id)+1``. 

483. Return a summary dict + a ``TaxonomyIdMaps`` snapshot the caller 

49 needs to hand to any import-mapping rebuild step. 

50"""

214def _category_group_lookup() -> dict[str, str]: 

215 """Return {category_name: group_title}. Each category belongs to exactly one group.""" 

216 out: dict[str, str] = {} 

217 for group_title, cats in ENTRY_GROUPS: 

218 for c in cats: 

219 if c in out: 

220 msg = f"Category {c!r} is listed in multiple groups in ENTRY_GROUPS" 

221 raise ValueError(msg) 

222 out[c] = group_title 

223 return out

231def _next_id(con: sqlite3.Connection, table: str) -> int: 

232 """Return the next available integer id for a catalog table.""" 

233 row = con.execute(f"SELECT COALESCE(MAX(id), 0) FROM {table}").fetchone() # noqa: S608 

234 return int(row[0]) + 1 if row else 1

237def _upsert_category_group( 

238 con: sqlite3.Connection, 

239 *, 

240 name: str, 

241 sort_order: int, 

242) -> int: 

243 """UPSERT a category_group by natural key (name); return its stable id.""" 

244 row = con.execute( 

245 "SELECT id FROM category_groups WHERE name = ?", 

246 [name], 

247 ).fetchone() 

248 if row is not None: 

249 gid = int(row[0]) 

250 con.execute( 

251 "UPDATE category_groups SET sort_order = ?, is_active = TRUE WHERE id = ?", 

252 [sort_order, gid], 

253 ) 

254 return gid 

255 gid = _next_id(con, "category_groups") 

256 con.execute( 

257 "INSERT INTO category_groups (id, name, sort_order, is_active) VALUES (?, ?, ?, TRUE)", 

258 [gid, name, sort_order], 

259 ) 

260 return gid

263def _upsert_category( 

264 con: sqlite3.Connection, 

265 *, 

266 name: str, 

267 group_id: int, 

268) -> int: 

269 """UPSERT a category by natural key (name); return its stable id. 

270 

271 A single ``UPDATE`` is enough even when child rows in ``expenses`` 

272 or mapping tables reference this category's id — updates on 

273 non-key columns are FK-safe. 

274 """ 

275 row = con.execute( 

276 "SELECT id, group_id, is_active FROM categories WHERE name = ?", 

277 [name], 

278 ).fetchone() 

279 if row is None: 

280 cid = _next_id(con, "categories") 

281 con.execute( 

282 "INSERT INTO categories (id, name, group_id, is_active) VALUES (?, ?, ?, TRUE)", 

283 [cid, name, group_id], 

284 ) 

285 return cid 

286 

287 cid = int(row[0]) 

288 existing_group = int(row[1]) if row[1] is not None else None 

289 existing_active = bool(row[2]) 

290 

291 if existing_group != group_id or not existing_active: 

292 con.execute( 

293 "UPDATE categories SET group_id = ?, is_active = TRUE WHERE id = ?", 

294 [group_id, cid], 

295 ) 

296 return cid

299def _upsert_event( # noqa: PLR0913 

300 con: sqlite3.Connection, 

301 *, 

302 name: str, 

303 date_from: date, 

304 date_to: date, 

305 auto_attach_enabled: bool, 

306 auto_tags: tuple[str, ...] | list[str] = (), 

307) -> int: 

308 """UPSERT an event by natural key (name); return its stable id. 

309 

310 ``auto_tags`` is stored as a JSON array on ``events.auto_tags``. 

311 The names must match ``tags.name`` rows at write time (runtime + 

312 historical write paths look them up by name; ``is_active = FALSE`` 

313 on a tag is a "hide from the ручной пикер" affordance and does 

314 not block event-driven auto-attach). This helper does not validate 

315 that — seed explicitly orders tag upserts before event upserts so 

316 the invariant holds. 

317 """ 

318 auto_tags_json = json.dumps(list(auto_tags), ensure_ascii=False) 

319 row = con.execute("SELECT id FROM events WHERE name = ?", [name]).fetchone() 

320 if row is not None: 

321 eid = int(row[0]) 

322 con.execute( 

323 "UPDATE events SET date_from = ?, date_to = ?, auto_attach_enabled = ?," 

324 " auto_tags = ?, is_active = TRUE WHERE id = ?", 

325 [date_from, date_to, auto_attach_enabled, auto_tags_json, eid], 

326 ) 

327 return eid 

328 eid = _next_id(con, "events") 

329 con.execute( 

330 "INSERT INTO events" 

331 " (id, name, date_from, date_to, auto_attach_enabled, is_active, auto_tags)" 

332 " VALUES (?, ?, ?, ?, ?, TRUE, ?)", 

333 [eid, name, date_from, date_to, auto_attach_enabled, auto_tags_json], 

334 ) 

335 return eid

338def _upsert_tag(con: sqlite3.Connection, *, name: str) -> int: 

339 """UPSERT a tag by natural key (name); return its stable id.""" 

340 row = con.execute("SELECT id FROM tags WHERE name = ?", [name]).fetchone() 

341 if row is not None: 

342 tid = int(row[0]) 

343 con.execute("UPDATE tags SET is_active = TRUE WHERE id = ?", [tid]) 

344 return tid 

345 tid = _next_id(con, "tags") 

346 con.execute( 

347 "INSERT INTO tags (id, name, is_active) VALUES (?, ?, TRUE)", 

348 [tid, name], 

349 ) 

350 return tid

358def seed_classification_catalog( # noqa: PLR0915 

359 con: sqlite3.Connection, 

360 *, 

361 year: int | None = None, 

362) -> tuple[dict, TaxonomyIdMaps]: 

363 """FK-safe in-place sync of the runtime taxonomy into ``dinary.db``. 

364 

365 Seeding order: deactivate-all -> category_groups -> categories -> 

366 tags -> events. Integer ids for pre-existing vocabulary are 

367 preserved; new vocabulary gets a fresh ``max(id)+1``. Rows present 

368 in the DB but absent from the new taxonomy snapshot stay 

369 ``is_active=FALSE`` so ledger rows keep a valid FK target while 

370 the live API hides them. 

371 

372 Does NOT touch ``import_mapping`` / ``import_mapping_tags`` — 

373 those tables are owned by ``imports.seed._rebuild_import_mapping`` 

374 and only relevant to deployments that actually import from 

375 Google Sheets. Non-import deployments (``inv bootstrap-catalog``, 

376 which calls ``bootstrap_catalog()`` below) never populate them. 

377 

378 Returns ``(summary_dict, TaxonomyIdMaps)``. The summary includes 

379 counts of each catalog table; the id-maps expose the ``name -> id`` 

380 lookups that ``_rebuild_import_mapping`` needs. 

381 """ 

382 if year is None: 

383 year = date.today().year 

384 

385 # 0. Deactivate everything; subsequent steps flip back active rows 

386 # in the new taxonomy snapshot. Rows that don't reappear stay 

387 # inactive and ledger FKs remain valid. 

388 con.execute("UPDATE category_groups SET is_active = FALSE") 

389 con.execute("UPDATE categories SET is_active = FALSE") 

390 con.execute("UPDATE events SET is_active = FALSE") 

391 con.execute("UPDATE tags SET is_active = FALSE") 

392 

393 # 1. category_groups (stable ids by name) 

394 group_id_by_title: dict[str, int] = {} 

395 for sort_order, (title, _cats) in enumerate(ENTRY_GROUPS, start=1): 

396 group_id_by_title[title] = _upsert_category_group( 

397 con, 

398 name=title, 

399 sort_order=sort_order, 

400 ) 

401 

402 # 2. categories (stable ids by name) 

403 cat_id_by_name: dict[str, int] = {} 

404 cat_to_group = _category_group_lookup() 

405 for cat_name in cat_to_group: 

406 cat_id_by_name[cat_name] = _upsert_category( 

407 con, 

408 name=cat_name, 

409 group_id=group_id_by_title[cat_to_group[cat_name]], 

410 ) 

411 # Also expose retired categories (is_active=FALSE) through the 

412 # in-memory map so any stray seed rule that names them can resolve 

413 # to the existing id rather than silently failing. Retired rows 

414 # deliberately keep their ids so mapping rebuilds can point at 

415 # them if a rule explicitly does (tests cover the rename path). 

416 for r in fetchall_as(ledger_repo.IdNameRow, con, load_sql("seed_load_categories.sql")): 

417 cat_id_by_name.setdefault(r.name, r.id) 

418 

419 # 3. tags (stable ids by name). Upserted BEFORE events so that 

420 # auto_tags names stored on ``events.auto_tags`` reliably resolve 

421 # to live ``tags.name`` rows on every subsequent lookup. 

422 tag_id_by_name: dict[str, int] = {} 

423 for tag_name in PHASE1_TAGS: 

424 tag_id_by_name[tag_name] = _upsert_tag(con, name=tag_name) 

425 

426 # 4. events (stable ids by name). 

427 event_id_by_name: dict[str, int] = {} 

428 vacation_auto_tags = tuple(VACATION_AUTO_TAGS) 

429 for y in range(HISTORICAL_YEAR_FROM, VACATION_EVENT_YEAR_TO + 1): 

430 name = f"{SYNTHETIC_EVENT_PREFIX}{y}" 

431 date_to_value = VACATION_EVENT_2026_END if y == VACATION_EVENT_YEAR_TO else date(y, 12, 31) 

432 event_id_by_name[name] = _upsert_event( 

433 con, 

434 name=name, 

435 date_from=date(y, 1, 1), 

436 date_to=date_to_value, 

437 auto_attach_enabled=True, 

438 auto_tags=vacation_auto_tags, 

439 ) 

440 for y in range(HISTORICAL_YEAR_FROM, BUSINESS_TRIP_EVENT_LAST_YEAR + 1): 

441 name = f"{BUSINESS_TRIP_EVENT_PREFIX}{y}" 

442 event_id_by_name[name] = _upsert_event( 

443 con, 

444 name=name, 

445 date_from=date(y, 1, 1), 

446 date_to=date(y, 12, 31), 

447 auto_attach_enabled=True, 

448 ) 

449 event_id_by_name[RELOCATION_EVENT_NAME] = _upsert_event( 

450 con, 

451 name=RELOCATION_EVENT_NAME, 

452 date_from=RELOCATION_EVENT_FROM, 

453 date_to=RELOCATION_EVENT_TO, 

454 auto_attach_enabled=False, 

455 ) 

456 for ev in EXPLICIT_EVENTS: 

457 event_id_by_name[ev.name] = _upsert_event( 

458 con, 

459 name=ev.name, 

460 date_from=ev.date_from, 

461 date_to=ev.date_to, 

462 auto_attach_enabled=ev.auto_attach_enabled, 

463 auto_tags=ev.auto_tags, 

464 ) 

465 

466 summary = { 

467 "category_groups": len(group_id_by_title), 

468 "categories": len(cat_id_by_name), 

469 "events": len(event_id_by_name), 

470 "tags": len(tag_id_by_name), 

471 } 

472 id_maps = TaxonomyIdMaps( 

473 group_id_by_title=group_id_by_title, 

474 cat_id_by_name=cat_id_by_name, 

475 tag_id_by_name=tag_id_by_name, 

476 event_id_by_name=event_id_by_name, 

477 ) 

478 return summary, id_maps

486def bootstrap_catalog(year: int | None = None) -> dict: 

487 """Non-destructive runtime-catalog bootstrap (no Google Sheets, no imports). 

488 

489 Seeds (and keeps in sync) the hardcoded Phase-1 taxonomy: 

490 ``category_groups``, ``categories``, ``tags``, ``events``. 

491 Idempotent and FK-safe — re-running never renumbers ids, never 

492 drops FK-referenced rows, and never mutates ``import_mapping`` 

493 / ``sheet_mapping``. 

494 

495 This is what ``inv bootstrap-catalog`` invokes on every deploy. 

496 Non-import users (``.deploy/import_sources.json`` absent or empty) 

497 rely on this as their only catalog-population path; import users 

498 get it as the first step of ``inv import-catalog`` too. 

499 

500 Does NOT bump ``app_metadata.catalog_version`` — that's reserved 

501 for ``rebuild_config_from_sheets`` (``inv import-catalog``) so 

502 routine bootstraps don't churn PWA caches. The very first run on 

503 a fresh DB still observes a version change because the row 

504 transitions from the schema-default ``0`` to the initial post-seed 

505 state; subsequent runs are proper no-ops. 

506 """ 

507 ledger_repo.init_db() 

508 

509 con = ledger_repo.get_connection() 

510 try: 

511 con.execute("BEGIN IMMEDIATE") 

512 try: 

513 summary, _ = seed_classification_catalog(con, year=year) 

514 con.execute("COMMIT") 

515 except Exception: 

516 ledger_repo.best_effort_rollback(con, context="bootstrap_catalog") 

517 raise 

518 finally: 

519 con.close() 

520 

521 logger.info("bootstrap_catalog complete: %s", summary) 

522 return summary

530def _bump_catalog_version(con: sqlite3.Connection, *, previous: int) -> int: 

531 """Increment ``catalog_version`` on the seed path (``inv import-catalog``). 

532 

533 One of the two write paths that touch ``catalog_version`` — the 

534 other is ``catalog_writer._commit_with_bump`` on the admin-API 

535 path. Both funnel through ``ledger_repo.set_catalog_version`` so 

536 future auditing hooks can intercept writes uniformly. See 

537 ``.plans/architecture.md`` §Catalog versioning. 

538 """ 

539 new_version = previous + 1 

540 ledger_repo.set_catalog_version(con, new_version) 

541 return new_version
