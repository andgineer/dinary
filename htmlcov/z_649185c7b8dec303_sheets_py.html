74def load_categories(ws: gspread.Worksheet | None = None) -> list[Category]: 

75 """Read unique (category, group) pairs from the sheet. 

76 

77 Scans every data row (skipping the header) and deduplicates by 

78 (category, group), preserving first-seen order. 

79 """ 

80 if ws is None: 

81 ws = get_sheet().sheet1 

82 

83 all_values = ws.get_all_values() 

84 seen: set[tuple[str, str]] = set() 

85 categories: list[Category] = [] 

86 

87 for row in all_values[HEADER_ROWS:]: 

88 cat_name = _cell(row, COL_CATEGORY) 

89 group_name = _cell(row, COL_GROUP) 

90 if cat_name and (cat_name, group_name) not in seen: 

91 seen.add((cat_name, group_name)) 

92 categories.append(Category(name=cat_name, group=group_name)) 

93 

94 _category_store.load(categories) 

95 logger.info("Loaded %d categories from sheet", len(categories)) 

96 return categories

114def find_category_row( 

115 all_values: list[list[str]], 

116 target_month: int, 

117 category: str, 

118 group: str, 

119) -> int | None: 

120 """Find the 1-indexed row for a (month, category, group) triple.""" 

121 for i, row in enumerate(all_values[HEADER_ROWS:], start=HEADER_ROWS + 1): 

122 month_val = _cell(row, COL_MONTH) 

123 cat_val = _cell(row, COL_CATEGORY) 

124 grp_val = _cell(row, COL_GROUP) 

125 if month_val == str(target_month) and cat_val == category and grp_val == group: 

126 return i 

127 return None

130def get_month_rate(all_values: list[list[str]], target_month: int) -> str | None: 

131 """Return the first non-empty EUR rate found for *target_month*.""" 

132 for row in all_values[HEADER_ROWS:]: 

133 if _cell(row, COL_MONTH) == str(target_month): 

134 rate_val = _cell(row, COL_RATE_EUR) 

135 if rate_val and _is_numeric(rate_val): 

136 return rate_val

140def find_month_range( 

141 all_values: list[list[str]], 

142 month: int, 

143) -> tuple[int, int] | None: 

144 """Return (first_row, last_row) 1-indexed for a contiguous month block.""" 

145 first: int | None = None 

146 last: int | None = None 

147 for i, row in enumerate(all_values[HEADER_ROWS:], start=HEADER_ROWS + 1): 

148 if _cell(row, COL_MONTH) == str(month): 

149 if first is None: 

150 first = i 

151 last = i 

152 if first is None or last is None: 

153 return None

157def create_month_rows( 

158 ws: gspread.Worksheet, 

159 all_values: list[list[str]], 

160 target: date, 

161) -> None: 

162 """Create rows for a new month by inserting after the header. 

163 

164 1. Find the previous month's rows as a template. 

165 2. Insert blank rows right after the header (row 2). 

166 3. CopyPaste the template into the new rows (source indices are 

167 shifted down by num_rows because of the insert). 

168 4. Clear amounts (B), comments (F), rate (H) and set the new date (A). 

169 """ 

170 target_month = target.month 

171 prev_month = target_month - 1 if target_month > 1 else 12 

172 

173 src = find_month_range(all_values, prev_month) 

174 if src is None: 

175 raise ValueError(f"No rows found for month {prev_month} to copy from") 

176 

177 src_start, src_end = src 

178 num_rows = src_end - src_start + 1 

179 dest_start = HEADER_ROWS + 1 # row 2 (right after header) 

180 

181 sheet_id = ws.id 

182 num_cols = len(all_values[0]) if all_values else COL_RATE_EUR 

183 

184 # After insertDimension the source rows shift down by num_rows 

185 shifted_src_start = src_start + num_rows 

186 

187 ws.spreadsheet.batch_update( 

188 { 

189 "requests": [ 

190 { 

191 "insertDimension": { 

192 "range": { 

193 "sheetId": sheet_id, 

194 "dimension": "ROWS", 

195 "startIndex": HEADER_ROWS, 

196 "endIndex": HEADER_ROWS + num_rows, 

197 }, 

198 "inheritFromBefore": False, 

199 }, 

200 }, 

201 { 

202 "copyPaste": { 

203 "source": { 

204 "sheetId": sheet_id, 

205 "startRowIndex": shifted_src_start - 1, 

206 "endRowIndex": shifted_src_start - 1 + num_rows, 

207 "startColumnIndex": 0, 

208 "endColumnIndex": num_cols, 

209 }, 

210 "destination": { 

211 "sheetId": sheet_id, 

212 "startRowIndex": dest_start - 1, 

213 "endRowIndex": dest_start - 1 + num_rows, 

214 "startColumnIndex": 0, 

215 "endColumnIndex": num_cols, 

216 }, 

217 "pasteType": "PASTE_NORMAL", 

218 }, 

219 }, 

220 ], 

221 }, 

222 ) 

223 

224 date_str = target.replace(day=1).strftime("%Y-%m-%d") 

225 batch: list[dict] = [] 

226 for i in range(num_rows): 

227 row = dest_start + i 

228 batch.extend( 

229 [ 

230 {"range": f"A{row}", "values": [[date_str]]}, 

231 {"range": f"B{row}", "values": [[""]]}, 

232 {"range": f"F{row}", "values": [[""]]}, 

233 {"range": f"H{row}", "values": [[""]]}, 

234 ], 

235 ) 

236 ws.batch_update(batch, value_input_option=ValueInputOption.user_entered) 

237 

238 logger.info("Created %d rows for month %d at top", num_rows, target_month)

264async def _ensure_rate( 

265 ws: gspread.Worksheet, 

266 all_values: list[list[str]], 

267 target_month: int, 

268 expense_date: date, 

269) -> Decimal: 

270 """Return the EUR/RSD rate for the month, fetching and storing it if absent.""" 

271 rate_str = get_month_rate(all_values, target_month) 

272 if rate_str: 

273 return Decimal(rate_str.replace(",", ".")) 

274 

275 rate = await fetch_eur_rsd_rate(expense_date.replace(day=1)) 

276 month_range = find_month_range(all_values, target_month)

279 logger.info("Wrote EUR/RSD rate %s for month %d at row %d", rate, target_month, rate_row)

283def append_to_rsd_formula(ws: gspread.Worksheet, row: int, amount_rsd: float) -> None: 

284 """Append +amount to the formula in column B (e.g. =460+373 → =460+373+1500).""" 

285 rsd_addr = gspread.utils.rowcol_to_a1(row, COL_AMOUNT_RSD) 

286 raw = ws.acell(rsd_addr, value_render_option=ValueRenderOption.formula).value 

287 existing = str(raw) if raw is not None else "" 

288 

289 amount_str = fmt_amount(amount_rsd) 

290 if existing.startswith("="): 

291 formula = f"{existing}+{amount_str}" 

292 elif existing and _is_numeric(existing): 

293 formula = f"={existing}+{amount_str}" 

294 else: 

295 formula = f"={amount_str}" 

296 

297 ws.update( 

298 range_name=rsd_addr, 

299 values=[[formula]], 

300 value_input_option=ValueInputOption.user_entered, 

301 )

311async def write_expense( 

312 amount_rsd: float, 

313 category: str, 

314 group: str, 

315 comment: str, 

316 expense_date: date, 

317) -> dict: 

318 """Append an expense to the Google Sheet. 

319 

320 Appends +amount to the formula in column B (RSD). 

321 Columns C (EUR) and G (month) are formula-driven and left untouched. 

322 """ 

323 ws = get_sheet().sheet1 

324 all_values = ws.get_all_values() 

325 target_month = expense_date.month 

326 

327 target_row = _resolve_row(ws, all_values, target_month, category, group, expense_date) 

328 

329 all_values = ws.get_all_values() 

330 row_data = all_values[target_row - 1] 

331 

332 rate = await _ensure_rate(ws, all_values, target_month, expense_date) 

333 append_to_rsd_formula(ws, target_row, amount_rsd) 

334 

335 if comment: 

336 append_comment(ws, target_row, row_data, comment) 

337 

338 existing_rsd = _cell(row_data, COL_AMOUNT_RSD) 

339 prev_total = float(existing_rsd.replace(",", ".")) if _is_numeric(existing_rsd) else 0.0 

340 eur_amount = float(Decimal(str(amount_rsd)) / rate) 

341 month_label = expense_date.strftime("%Y-%m") 

342 

343 logger.info( 

344 "Wrote expense: %s RSD (%.2f EUR) to %s/%s in %s", 

345 amount_rsd, 

346 eur_amount, 

347 category, 

348 group, 

349 month_label, 

350 ) 

351 

352 return { 

353 "month": month_label, 

354 "category": category, 

355 "amount_rsd": amount_rsd, 

356 "amount_eur": round(eur_amount, 2), 

357 "new_total_rsd": round(prev_total + amount_rsd, 2), 

358 }
