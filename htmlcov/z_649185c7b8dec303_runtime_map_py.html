1"""Runtime 3D -> 2D mapping sourced from a curated ``map`` worksheet tab. 

2 

3The map tab is a flat, human-editable table whose rows are evaluated 

4first-match-wins by the drain loop when it needs to pick a 

5``(sheet_category, sheet_group)`` target for an expense. 

6 

7Tab layout (columns A..E, one header row, then data rows): 

8 

9 A category canonical category name (must match an active 

10 categories.name) 

11 B event_pattern fnmatch-style glob against the expense's event 

12 name ("" -> match any event or no event; 

13 "отпуск-*" -> any vacation year; "evt" -> 

14 exact match) 

15 C tags comma- or space-separated list of tag names 

16 required on the expense ("" -> no tags required; 

17 expenses may carry extra tags) 

18 D sheet_category target column D value on the logging sheet 

19 E sheet_group target column E value on the logging sheet 

20 

21More-specific rules live near the top; generic catch-alls live near 

22the bottom. Row ordering in the sheet maps directly to 

23``runtime_mapping.row_order`` in the DB. 

24 

25Reload model 

26------------ 

27 

28The DB side (``runtime_mapping`` + ``runtime_mapping_tags``) is 

29considered derived state; the tab is the source of truth. This module 

30exposes two entry points for refreshing the DB side: 

31 

32* ``ensure_fresh()`` -- cheap modifiedTime check via Drive API; the 

33 drain loop calls this before each batch. Re-parses only when the 

34 tab has actually changed since the last successful reload. 

35* ``reload_now()`` -- unconditional reload, used by the admin reload 

36 endpoint. 

37 

38Both swap the DB tables atomically inside a single transaction so the 

39drain worker never observes a half-populated map. 

40"""

119def _case_insensitive_match_hint(missing: str, candidates: Iterable[str]) -> str: 

120 """Return a ``; did you mean 'X'?`` hint for case-only mismatches. 

121 

122 Helps the operator land on "the DB has 'еда', the map tab has 

123 'Еда'" without having to compare character by character in a 

124 Cyrillic string. Empty when no case-insensitive match exists; 

125 only suggests the first matching candidate to keep the error 

126 message short. 

127 """ 

128 lower = missing.casefold() 

129 for candidate in candidates: 

130 if candidate.casefold() == lower: 

131 return f"; did you mean {candidate!r}?" 

132 return ""

135def _parse_tags_cell(raw: str) -> list[str]: 

136 """Split a tags-cell value into a list of trimmed tag names.

145def _validate_event_pattern(pattern: str) -> None: 

146 """Reject patterns that would match literally nothing. 

147 

148 Empty pattern is legal ("any event or no event"). Any other pattern 

149 is passed through to ``fnmatch`` at resolve time; here we just 

150 sanity-check the fnmatch compile doesn't crash. 

151 """ 

152 if pattern == "": 

153 return 

154 try: 

155 fnmatch.translate(pattern) 

156 except Exception as exc: 

157 msg = f"Invalid event_pattern {pattern!r}: {exc}" 

158 raise MapTabError(msg) from exc

161def parse_rows( 

162 raw_rows: list[list[str]], 

163 *, 

164 cat_id_by_name: dict[str, int], 

165 tag_id_by_name: dict[str, int], 

166) -> list[MapRow]: 

167 """Validate and resolve raw ``map`` tab rows into ``MapRow`` objects. 

168 

169 Skips blank rows (every cell empty) so the operator can use empty 

170 rows as visual separators. Raises ``MapTabError`` on the first row 

171 whose category/tag name doesn't resolve against the active catalog. 

172 """ 

173 parsed: list[MapRow] = [] 

174 for idx, row in enumerate(raw_rows, start=1): 

175 # Pad to 5 columns for safety; gspread drops trailing empty cells. 

176 cells = list(row) + [""] * max(0, 5 - len(row)) 

177 cat_name, event_pattern, tags_cell, sheet_category, sheet_group = ( 

178 cells[0].strip(), 

179 cells[1].strip(), 

180 cells[2].strip(), 

181 cells[3].strip(), 

182 cells[4].strip(), 

183 ) 

184 

185 # Visual separator / blank line: every meaningful cell empty. 

186 if not any([cat_name, event_pattern, tags_cell, sheet_category, sheet_group]): 

187 continue 

188 

189 if not cat_name: 

190 msg = f"map tab row {idx}: category name is required" 

191 raise MapTabError(msg) 

192 if cat_name not in cat_id_by_name: 

193 hint = _case_insensitive_match_hint(cat_name, cat_id_by_name.keys()) 

194 msg = ( 

195 f"map tab row {idx}: category {cat_name!r} is not an active " 

196 f"categories.name (names are case-sensitive; was it renamed or retired?)" 

197 f"{hint}" 

198 ) 

199 raise MapTabError(msg) 

200 

201 if not sheet_category: 

202 msg = ( 

203 f"map tab row {idx} ({cat_name!r}): sheet_category (column D) " 

204 "must not be empty — it's the target cell on the logging sheet" 

205 ) 

206 raise MapTabError(msg) 

207 

208 _validate_event_pattern(event_pattern) 

209 

210 tag_names = _parse_tags_cell(tags_cell) 

211 tag_ids: list[int] = [] 

212 for tag_name in tag_names: 

213 if tag_name not in tag_id_by_name: 

214 hint = _case_insensitive_match_hint(tag_name, tag_id_by_name.keys()) 

215 msg = ( 

216 f"map tab row {idx} ({cat_name!r}): tag {tag_name!r} is not an " 

217 "active tags.name (names are case-sensitive; was it renamed or retired?)" 

218 f"{hint}" 

219 ) 

220 raise MapTabError(msg) 

221 tag_ids.append(tag_id_by_name[tag_name]) 

222 

223 parsed.append( 

224 MapRow( 

225 row_order=len(parsed) + 1, 

226 category_id=cat_id_by_name[cat_name], 

227 event_pattern=event_pattern, 

228 tag_ids=tuple(sorted(set(tag_ids))), 

229 sheet_category=sheet_category, 

230 sheet_group=sheet_group, 

231 ), 

232 ) 

233 return parsed

241def _load_active_catalog( 

242 con: duckdb.DuckDBPyConnection, 

243) -> tuple[dict[str, int], dict[str, int]]: 

244 cat_rows = con.execute( 

245 "SELECT name, id FROM categories WHERE is_active", 

246 ).fetchall() 

247 tag_rows = con.execute( 

248 "SELECT name, id FROM tags WHERE is_active", 

249 ).fetchall() 

250 cat_id_by_name = {str(r[0]): int(r[1]) for r in cat_rows} 

251 tag_id_by_name = {str(r[0]): int(r[1]) for r in tag_rows} 

252 return cat_id_by_name, tag_id_by_name

255def _atomic_swap(con: duckdb.DuckDBPyConnection, rows: list[MapRow]) -> None: 

256 """Wipe runtime_mapping(_tags) and repopulate from ``rows``. 

257 

258 Single transaction so the drain loop never sees a half-populated 

259 table. DuckDB's FK-in-transaction behaviour (see 

260 ``seed_config._purge_mapping_tables``) is OK here because the only 

261 FK points from ``runtime_mapping_tags`` into ``runtime_mapping``, 

262 which we delete in the correct order. 

263 """ 

264 con.execute("BEGIN") 

265 try: 

266 con.execute("DELETE FROM runtime_mapping_tags") 

267 con.execute("DELETE FROM runtime_mapping") 

268 for row in rows: 

269 con.execute( 

270 "INSERT INTO runtime_mapping" 

271 " (row_order, category_id, event_pattern, sheet_category, sheet_group)" 

272 " VALUES (?, ?, ?, ?, ?)", 

273 [ 

274 row.row_order, 

275 row.category_id, 

276 row.event_pattern, 

277 row.sheet_category, 

278 row.sheet_group, 

279 ], 

280 ) 

281 for tag_id in row.tag_ids: 

282 con.execute( 

283 "INSERT INTO runtime_mapping_tags (mapping_row_order, tag_id) VALUES (?, ?)", 

284 [row.row_order, tag_id], 

285 ) 

286 con.execute("COMMIT") 

287 except Exception: 

288 duckdb_repo.best_effort_rollback(con, context="runtime_map._atomic_swap") 

289 raise

297def reload_now(*, check_after: bool = True) -> dict: 

298 """Unconditional reload: fetch the map tab, parse, swap the DB. 

299 

300 Returns a summary dict with the new ``modifiedTime``, number of 

301 parsed rows, and the per-category row counts. Raises 

302 ``MapTabError`` on parse/validation failure without touching the 

303 DB; the current ``runtime_mapping`` contents stay in place. 

304 

305 ``check_after`` controls the lost-update guard (two Drive 

306 metadata GETs): 

307 

308 * ``True`` (default, drain-loop path): capture ``modifiedTime`` 

309 both before and after reading the rows. When they agree, the 

310 tab was stable during the read and we cache the value; when 

311 they differ, the parsed rows pre-date an edit, so we install 

312 them anyway (they're at least as fresh as what was in the DB) 

313 but leave the cache unset so the next ``ensure_fresh`` tick 

314 retries and converges. 

315 * ``False`` (admin-reload path): single Drive GET before the 

316 read, cache it unconditionally. The operator explicitly asked 

317 for a reload, so saving one Drive round-trip matters more than 

318 the self-healing retry — if they edit again, they click 

319 reload again. This halves the Drive quota spend on the hot 

320 admin button, which matters because Drive's metadata quota is 

321 lower than the Sheets one. 

322 """ 

323 spreadsheet_id = settings.sheet_logging_spreadsheet 

324 if not spreadsheet_id: 

325 msg = "sheet_logging_spreadsheet not configured; nothing to reload" 

326 raise MapTabError(msg) 

327 

328 # Capture modifiedTime BEFORE the content read so a concurrent 

329 # edit that bumps modifiedTime while we're fetching rows can be 

330 # detected by the optional second GET below. 

331 modified_time_before = drive_get_modified_time(spreadsheet_id) 

332 

333 sh = get_sheet(spreadsheet_id) 

334 try: 

335 ws = sh.worksheet(settings.runtime_map_tab_name) 

336 except gspread.WorksheetNotFound as exc: 

337 msg = ( 

338 f"map tab {settings.runtime_map_tab_name!r} not found on " 

339 f"spreadsheet {spreadsheet_id!r}; create it via " 

340 "ensure_default_map_tab() or copy the template from docs/" 

341 ) 

342 raise MapTabError(msg) from exc 

343 

344 # Skip one header row; tolerate an arbitrary number of data rows. 

345 raw = ws.get_all_values()[1:] 

346 

347 con = duckdb_repo.get_connection() 

348 try: 

349 cat_id_by_name, tag_id_by_name = _load_active_catalog(con) 

350 rows = parse_rows( 

351 raw, 

352 cat_id_by_name=cat_id_by_name, 

353 tag_id_by_name=tag_id_by_name, 

354 ) 

355 _atomic_swap(con, rows) 

356 finally: 

357 con.close() 

358 

359 global _last_seen_modified_time # noqa: PLW0603 

360 if not check_after: 

361 # Admin-initiated reload: cache eagerly, skip the second GET. 

362 _last_seen_modified_time = modified_time_before 

363 modified_time_after = modified_time_before 

364 cached = True 

365 else: 

366 modified_time_after = drive_get_modified_time(spreadsheet_id) 

367 if modified_time_after == modified_time_before: 

368 _last_seen_modified_time = modified_time_after 

369 cached = True 

370 else: 

371 logger.info( 

372 "runtime_map: modifiedTime shifted during reload (%s -> %s); " 

373 "leaving cache unset so the next ensure_fresh() retries", 

374 modified_time_before, 

375 modified_time_after, 

376 ) 

377 cached = False 

378 

379 summary: dict = { 

380 "row_count": len(rows), 

381 "modified_time": modified_time_after, 

382 "tab": settings.runtime_map_tab_name, 

383 "modified_time_cached": cached, 

384 } 

385 logger.info("runtime_map reloaded: %s", summary) 

386 return summary

389def ensure_fresh() -> None: 

390 """Drain-loop hook: reload the map tab iff Drive says it changed. 

391 

392 Cheap path (steady state): one Drive metadata GET, sub-second, 

393 compares a cached RFC3339 timestamp. Reload path (tab edited): 

394 one Sheets ``get_all_values()`` call plus a DB swap. 

395 

396 Network errors downgrade to a log warning without raising — the 

397 drain loop must keep draining whatever is in ``runtime_mapping`` 

398 rather than stalling on an unrelated Google outage. 

399 """ 

400 spreadsheet_id = settings.sheet_logging_spreadsheet 

401 if not spreadsheet_id: 

402 return 

403 try: 

404 modified_time = drive_get_modified_time(spreadsheet_id) 

405 except Exception as exc: # noqa: BLE001 

406 # Drive client raises from a large surface (auth expiry, 

407 # network errors, 5xx, JSON decode). The drain loop must 

408 # keep draining whatever is in ``runtime_mapping`` rather 

409 # than stall on an unrelated Google outage, so we downgrade 

410 # *any* exception to a warning and bail out of this tick. 

411 logger.warning( 

412 "runtime_map: drive_get_modified_time failed (%s); keeping cached runtime_mapping", 

413 exc.__class__.__name__, 

414 ) 

415 return 

416 if modified_time == _last_seen_modified_time: 

417 return 

418 logger.info( 

419 "runtime_map: map-tab changed (%s -> %s); reloading", 

420 _last_seen_modified_time, 

421 modified_time, 

422 ) 

423 try: 

424 reload_now() 

425 except Exception: 

426 # Keep the last-known-good runtime_mapping; surface the problem 

427 # in logs for the operator. Intentionally don't update 

428 # _last_seen_modified_time on failure so the next drain pass 

429 # retries. 

430 logger.exception("runtime_map: reload_now() failed; keeping cached mapping")

433def _warn_if_existing_map_tab_is_stale( 

434 ws: gspread.Worksheet, 

435 con: duckdb.DuckDBPyConnection, 

436) -> None: 

437 """Emit a WARN log if the existing ``map`` tab references names 

438 no longer in the active catalog. 

439 

440 Called from ``ensure_default_map_tab`` when the tab already 

441 exists. Silence (tab exists → return) used to leave the operator 

442 guessing whether a reseed that renamed/retired categories had 

443 broken the runtime mapping; the drain loop would only surface 

444 the problem on the next job via ``MapTabError``. Running the 

445 parser here as a dry-run lets the reseed command log the problem 

446 immediately without mutating the tab (the operator is the one 

447 who should pick the replacement cell, not us). 

448 

449 Failures in the dry-run are intentionally non-fatal: the seed 

450 task has already committed the catalog change and we don't want 

451 to roll it back just because the map tab is out of date. 

452 """ 

453 try: 

454 raw = ws.get_all_values()[1:] 

455 cat_id_by_name, tag_id_by_name = _load_active_catalog(con) 

456 parse_rows( 

457 raw, 

458 cat_id_by_name=cat_id_by_name, 

459 tag_id_by_name=tag_id_by_name, 

460 ) 

461 except MapTabError as exc: 

462 logger.warning( 

463 "ensure_default_map_tab: existing %r tab references names not in the " 

464 "current active catalog — runtime logging will fail until the tab is " 

465 "fixed: %s", 

466 settings.runtime_map_tab_name, 

467 exc, 

468 ) 

469 except Exception: 

470 # Tab read itself failed (network / permissions) — surface in 

471 # logs but don't block reseed completion. 

472 logger.exception( 

473 "ensure_default_map_tab: could not validate existing %r tab; skipping staleness check", 

474 settings.runtime_map_tab_name, 

475 )

478def ensure_default_map_tab() -> None: 

479 """Create the ``map`` worksheet tab with a default layout if it doesn't exist. 

480 

481 Per-category default: prefer the ``(sheet_category, sheet_group)`` 

482 pair the historical importer resolved for the **latest** 

483 ``import_sources.year`` — that's the pair the operator has been 

484 using in the actual logging sheet, so the default mapping produces 

485 cells that already exist on the sheet (column D labels in the 

486 legacy sheet are often mixed-case localised forms like ``"Машина"`` 

487 that don't match the canonical 3D name ``"машина"``). If no 

488 historical mapping exists for a category (e.g. a brand-new 

489 taxonomy entry), fall back to the 3D category name in both A and 

490 D and an empty group in E; the drain-loop category-name fallback 

491 in ``sheet_logging._drain_one_job`` takes care of the rest. 

492 

493 Idempotent: does nothing when the tab already exists. When it 

494 does exist, though, we dry-run the parser against the current 

495 active catalog and WARN if any row references names that have 

496 been renamed or retired since the tab was last curated — the 

497 operator needs to know immediately rather than waiting for the 

498 drain loop to surface ``MapTabError`` at runtime. Safe to call 

499 from seed and from the admin API. 

500 """ 

501 spreadsheet_id = settings.sheet_logging_spreadsheet 

502 if not spreadsheet_id: 

503 logger.info("ensure_default_map_tab: sheet_logging_spreadsheet empty; skipping") 

504 return 

505 sh = get_sheet(spreadsheet_id) 

506 try: 

507 existing_ws = sh.worksheet(settings.runtime_map_tab_name) 

508 except gspread.WorksheetNotFound: 

509 pass 

510 else: 

511 con = duckdb_repo.get_connection() 

512 try: 

513 _warn_if_existing_map_tab_is_stale(existing_ws, con) 

514 finally: 

515 con.close() 

516 return 

517 

518 con = duckdb_repo.get_connection() 

519 try: 

520 cat_rows = con.execute( 

521 "SELECT c.id, c.name FROM categories c" 

522 " JOIN category_groups g ON g.id = c.group_id" 

523 " WHERE c.is_active AND g.is_active" 

524 " ORDER BY g.sort_order, c.name", 

525 ).fetchall() 

526 latest_year_row = con.execute( 

527 "SELECT MAX(year) FROM import_sources", 

528 ).fetchone() 

529 latest_year = ( 

530 int(latest_year_row[0]) 

531 if latest_year_row is not None and latest_year_row[0] is not None 

532 else None 

533 ) 

534 # Historical 2D labels per category_id, keyed to the latest 

535 # imported year (or year=0 fallback entries). A category may 

536 # appear under several legacy ``(sheet_category, sheet_group)`` 

537 # pairs; we pick the first one deterministically so the seed 

538 # output is stable. The operator is expected to curate after. 

539 latest_pairs: dict[int, tuple[str, str]] = {} 

540 if latest_year is not None: 

541 for r in con.execute( 

542 "SELECT category_id, sheet_category, sheet_group" 

543 " FROM import_mapping" 

544 " WHERE year IN (?, 0)" 

545 " ORDER BY year DESC, id", 

546 [latest_year], 

547 ).fetchall(): 

548 cid = int(r[0]) 

549 if cid in latest_pairs: 

550 continue 

551 latest_pairs[cid] = (str(r[1]), str(r[2] or "")) 

552 finally: 

553 con.close() 

554 

555 header = ["category", "event_pattern", "tags", "sheet_category", "sheet_group"] 

556 body: list[list[str]] = [] 

557 identity_fallbacks = 0 

558 for cid, cname in cat_rows: 

559 cname_s = str(cname) 

560 pair = latest_pairs.get(int(cid)) 

561 if pair is None: 

562 identity_fallbacks += 1 

563 body.append([cname_s, "", "", cname_s, ""]) 

564 else: 

565 body.append([cname_s, "", "", pair[0], pair[1]]) 

566 values = [header, *body] 

567 

568 ws = sh.add_worksheet( 

569 title=settings.runtime_map_tab_name, 

570 rows=max(len(values) + 10, 50), 

571 cols=len(header), 

572 ) 

573 ws.update(range_name="A1", values=values) 

574 logger.info( 

575 "ensure_default_map_tab: created %r with %d rows " 

576 "(%d seeded from import_mapping year=%s, %d identity fallbacks) — " 

577 "please review the tab before relying on runtime logging", 

578 settings.runtime_map_tab_name, 

579 len(body), 

580 len(body) - identity_fallbacks, 

581 latest_year, 

582 identity_fallbacks, 

583 )
