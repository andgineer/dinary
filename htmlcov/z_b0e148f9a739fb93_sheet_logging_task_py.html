13async def warm_sheet_mapping() -> None: 

14 """Preload ``sheet_mapping`` from the ``map`` tab at startup. 

15 

16 Moves the ~1s Drive+Sheets round-trip off the first-expense hot 

17 path. Skipped when the drain loop itself is disabled (interval 

18 <= 0 — the test fixture uses this to avoid network I/O in 

19 lifespan) or when sheet-logging is unconfigured, or when the 

20 operator explicitly disabled the warm-up via 

21 ``warm_sheet_mapping_timeout_sec <= 0``. 

22 

23 Bounded by ``settings.warm_sheet_mapping_timeout_sec`` so a slow 

24 or unreachable Google backend cannot wedge the lifespan startup 

25 and delay ``/api/health`` from answering (a long startup starves 

26 Railway's health probe). On timeout or any other failure we log 

27 and continue — the drain loop's ``ensure_fresh`` will retry on 

28 its own schedule, and the cached mapping (from the last 

29 successful reload before the restart) keeps the runtime 

30 functional in the meantime. 

31 

32 Note on cancellation: ``asyncio.wait_for`` cancels the awaitable 

33 but cannot cancel the underlying ``to_thread`` worker (sync 

34 Google I/O). The worker continues running until its socket 

35 timeout fires; the event loop is free to move on immediately. 

36 """ 

37 if settings.sheet_logging_drain_interval_sec <= 0: 

38 return 

39 if not sheet_logging.is_sheet_logging_enabled(): 

40 return 

41 timeout = settings.warm_sheet_mapping_timeout_sec 

42 if timeout <= 0: 

43 return 

44 try: 

45 summary = await asyncio.wait_for( 

46 asyncio.to_thread(sheet_mapping.reload_now), 

47 timeout=timeout, 

48 ) 

49 logger.info("sheet_mapping preloaded at startup: %s", summary) 

50 except TimeoutError: 

51 logger.warning( 

52 "sheet_mapping preload timed out after %.1fs; " 

53 "drain loop will retry on its own schedule", 

54 timeout, 

55 ) 

56 except Exception: 

57 logger.exception("sheet_mapping preload failed; drain loop will retry")

60async def sheet_logging_task() -> None: 

61 interval = settings.sheet_logging_drain_interval_sec 

62 enabled = sheet_logging.is_sheet_logging_enabled() 

63 if not enabled or interval <= 0: 

64 logger.info( 

65 "sheet-logging task disabled (interval<=0 or DINARY_SHEET_LOGGING_SPREADSHEET unset)", 

66 ) 

67 return 

68 # Register a wake-up event so producers (e.g. POST /api/expenses) 

69 # can kick a sweep immediately instead of waiting for the next 

70 # periodic tick. The timer is still the fallback for crash-recovery 

71 # sweeps over jobs left behind by a previous worker. 

72 wake = asyncio.Event() 

73 loop = asyncio.get_running_loop() 

74 sheet_logging.register_wake_channel(wake, loop) 

75 logger.info("sheet-logging task started: interval=%gs", interval) 

76 try: 

77 first = True 

78 while True: 

79 if not first: 

80 with contextlib.suppress(TimeoutError): 

81 await asyncio.wait_for(wake.wait(), timeout=interval) 

82 wake.clear() 

83 first = False 

84 try: 

85 summary = await asyncio.to_thread(sheet_logging.drain_pending) 

86 attempted = summary.get("attempted", 0) 

87 cap_reached = summary.get("cap_reached", False) 

88 poisoned = summary.get("poisoned", 0) 

89 failed = summary.get("failed", 0) 

90 if attempted > 0 or cap_reached or poisoned > 0 or failed > 0: 

91 logger.info("drain sweep: %s", summary) 

92 else: 

93 logger.debug("drain sweep: %s", summary) 

94 except asyncio.CancelledError: 

95 raise 

96 except Exception: 

97 logger.exception("drain sweep failed") 

98 finally: 

99 sheet_logging.clear_wake_channel()
