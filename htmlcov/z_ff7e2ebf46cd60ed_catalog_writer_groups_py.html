21def _group_usage_count(con: sqlite3.Connection, group_id: int) -> int: 

22 """Active child count for observability in ``edit_group``. 

23 

24 Distinct from ``_group_child_category_count`` (used by 

25 ``delete_group``) which counts *all* children, active or not. The 

26 distinction is intentional: 

27 

28 * ``edit_group(is_active=False)`` asks "is this group still 

29 relevant to the operator?" — an inactive category is already 

30 hidden from new expenses, so it should not block flipping the 

31 parent inactive. 

32 * ``delete_group`` asks "is it safe to physically remove this row?" 

33 — FK from ``categories.group_id`` does not care about 

34 ``is_active`` so any surviving category row, active or not, will 

35 cause a constraint violation. 

36 """ 

37 row = con.execute( 

38 "SELECT COUNT(*) FROM categories WHERE group_id = ? AND is_active", 

39 [group_id], 

40 ).fetchone() 

41 return int(row[0]) if row else 0

44def _group_child_category_count( 

45 con: sqlite3.Connection, 

46 group_id: int, 

47) -> int: 

48 """Total categories (active or inactive) pointing at this group. 

49 

50 Used by ``delete_group`` to gate physical removal: a non-zero 

51 count means the ``categories.group_id`` FK would reject the 

52 ``DELETE``, so the operator has to relocate or delete the 

53 children first. 

54 """ 

55 row = con.execute( 

56 "SELECT COUNT(*) FROM categories WHERE group_id = ?", 

57 [group_id], 

58 ).fetchone() 

59 return int(row[0]) if row else 0

62def add_group( 

63 con: sqlite3.Connection, 

64 *, 

65 name: str, 

66 sort_order: int | None = None, 

67) -> AddResult: 

68 """Create a new category group, or reactivate-in-place if the name exists. 

69 

70 Reactivate preserves the existing ``sort_order`` (see 

71 ``catalog_writer`` module docstring for the reactivate contract). 

72 To change ``sort_order``, use ``edit_group``. 

73 """ 

74 con.execute("BEGIN IMMEDIATE") 

75 try: 

76 before = hash_state(con) 

77 existing = con.execute( 

78 "SELECT id, is_active FROM category_groups WHERE name = ?", 

79 [name], 

80 ).fetchone() 

81 if existing is not None: 

82 gid = int(existing[0]) 

83 was_active = bool(existing[1]) 

84 if not was_active: 

85 con.execute( 

86 "UPDATE category_groups SET is_active = TRUE WHERE id = ?", 

87 [gid], 

88 ) 

89 bumped = commit_with_bump( 

90 con, 

91 before, 

92 context=f"add_group(reactivate name={name!r})", 

93 ) 

94 status: AddStatus = "reactivated" if bumped else "noop" 

95 return AddResult(id=gid, status=status) 

96 gid = next_id(con, "category_groups") 

97 if sort_order is None: 

98 row = con.execute( 

99 "SELECT COALESCE(MAX(sort_order), 0) + 1 FROM category_groups", 

100 ).fetchone() 

101 sort_order = int(row[0]) if row else 1 

102 con.execute( 

103 "INSERT INTO category_groups (id, name, sort_order, is_active) VALUES (?, ?, ?, TRUE)", 

104 [gid, name, sort_order], 

105 ) 

106 commit_with_bump(con, before, context=f"add_group(name={name!r})") 

107 return AddResult(id=gid, status="created") 

108 except Exception: 

109 storage.best_effort_rollback(con, context="catalog_writer.add_group") 

110 raise

113def edit_group( 

114 con: sqlite3.Connection, 

115 group_id: int, 

116 *, 

117 name: str | None = None, 

118 sort_order: int | None = None, 

119 is_active: bool | None = None, 

120) -> None: 

121 """Atomic PATCH for ``category_groups``. 

122 

123 All parameters optional; absent parameters are left unchanged. 

124 ``is_active=False`` enforces the same in-use guard as the legacy 

125 ``set_group_active``. Validations (not-found, conflict, in-use) 

126 run *before* any UPDATE so a failed validation never leaves the 

127 row half-edited even if the caller PATCHes multiple columns at 

128 once. 

129 """ 

130 con.execute("BEGIN IMMEDIATE") 

131 try: 

132 before = hash_state(con) 

133 row = con.execute( 

134 "SELECT id FROM category_groups WHERE id = ?", 

135 [group_id], 

136 ).fetchone() 

137 if row is None: 

138 raise CatalogNotFoundError(f"category_group id={group_id} not found") 

139 if name is not None: 

140 conflict = con.execute( 

141 "SELECT id FROM category_groups WHERE name = ? AND id != ?", 

142 [name, group_id], 

143 ).fetchone() 

144 if conflict is not None: 

145 raise CatalogConflictError( 

146 f"category_group name {name!r} already in use by id={int(conflict[0])}", 

147 ) 

148 if is_active is False: 

149 usage = _group_usage_count(con, group_id) 

150 if usage > 0: 

151 raise CatalogInUseError("category_group", group_id, usage) 

152 if name is not None: 

153 con.execute( 

154 "UPDATE category_groups SET name = ? WHERE id = ?", 

155 [name, group_id], 

156 ) 

157 if sort_order is not None: 

158 con.execute( 

159 "UPDATE category_groups SET sort_order = ? WHERE id = ?", 

160 [sort_order, group_id], 

161 ) 

162 if is_active is not None: 

163 con.execute( 

164 "UPDATE category_groups SET is_active = ? WHERE id = ?", 

165 [bool(is_active), group_id], 

166 ) 

167 commit_with_bump(con, before, context=f"edit_group(id={group_id})") 

168 except Exception: 

169 storage.best_effort_rollback(con, context="catalog_writer.edit_group") 

170 raise

182def delete_group( 

183 con: sqlite3.Connection, 

184 group_id: int, 

185) -> DeleteResult: 

186 """Hard-delete a category group iff it has no (active or inactive) 

187 categories referencing it; raise 409 otherwise. 

188 

189 Unlike the other catalog kinds, soft-deleting a group while 

190 categories still point at it leaves orphaned ``is_active=TRUE`` 

191 categories attached to an inactive group — the API refuses that 

192 combination outright and forces the operator to relocate (or 

193 delete) the categories first. 

194 """ 

195 con.execute("BEGIN IMMEDIATE") 

196 try: 

197 before = hash_state(con) 

198 row = con.execute( 

199 "SELECT id FROM category_groups WHERE id = ?", 

200 [group_id], 

201 ).fetchone() 

202 if row is None: 

203 raise CatalogNotFoundError(f"category_group id={group_id} not found") 

204 child_count = _group_child_category_count(con, group_id) 

205 if child_count > 0: 

206 raise CatalogInUseError("category_group", group_id, child_count) 

207 con.execute("DELETE FROM category_groups WHERE id = ?", [group_id]) 

208 commit_with_bump(con, before, context=f"delete_group(id={group_id})") 

209 return DeleteResult(status="hard", usage_count=0) 

210 except Exception: 

211 storage.best_effort_rollback(con, context="catalog_writer.delete_group") 

212 raise
