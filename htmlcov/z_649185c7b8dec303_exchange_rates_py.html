1"""Exchange rate service with SQLite storage. 

2 

3Fetches rates from kurs.resenje.org (NBS) when one of the currencies is RSD, 

4and from api.frankfurter.dev (ECB) otherwise. Rates are stored in the 

5``exchange_rates`` table in ``data/dinary.db`` as 

6``1 source_currency * rate = amount in target_currency``. 

7""" 

8 

9import logging 

10from datetime import date 

11from decimal import Decimal 

12 

13from cachetools import TTLCache, cached 

14 

15from dinary.services.nbs import _resolve_from_nbs 

16from dinary.services.rate_helpers import ( 

17 _FETCH_RATE_CACHE_TIME, 

18 _get_db_rate, 

19 _get_json_or_none, 

20 _get_latest_db_rate, 

21 _save_db_rate, 

22) 

23 

24logger = logging.getLogger(__name__) 

25 

26FRANKFURTER_URL = "https://api.frankfurter.dev/v1" 

27 

28 

29# --------------------------------------------------------------------------- 

30# Frankfurter fetcher (TTL-cached to avoid rate-limiting) 

31# --------------------------------------------------------------------------- 

32 

33 

34@cached(cache=TTLCache(maxsize=10000, ttl=_FETCH_RATE_CACHE_TIME)) 

35def _fetch_frankfurter_rate(rate_date: date, source: str, target: str) -> Decimal | None: 

36 """Fetch rate from Frankfurter (ECB): 1 unit of `source` = ? `target`. 

37 

38 Frankfurter returns the nearest previous working day when the given date 

39 is a weekend/holiday, so we don't need our own walk-back for it. 

40 

41 ``None`` (HTTP failure) is intentionally cached for the full TTL so 

42 that a down Frankfurter service is not hammered with retries on every 

43 request. Callers fall back to DB rates when this returns ``None``. 

44 Do NOT "fix" this by skipping ``None`` caching. 

45 """ 

46 url = f"{FRANKFURTER_URL}/{rate_date.isoformat()}" 

47 data = _get_json_or_none(url, params={"base": source, "symbols": target}) 

48 if data: 

49 rates = data.get("rates") or {} 

50 if target in rates: 

51 logger.info("Got rate from frankfurter") 

52 return Decimal(str(rates[target])) 

53 return None 

54 

55 

56# --------------------------------------------------------------------------- 

57# Frankfurter resolution 

58# --------------------------------------------------------------------------- 

59 

60 

61def _resolve_from_frankfurter(con, rate_date: date, source: str, target: str) -> Decimal | None: 

62 """Fetch rate from Frankfurter. Falls back to last known DB rate on failure.""" 

63 rate = _fetch_frankfurter_rate(rate_date, source, target) 

64 if rate is not None: 

65 _save_db_rate(con, rate_date, source, target, rate) 

66 return rate 

67 fallback = _get_latest_db_rate(con, source, target) 

68 if fallback is not None: 

69 logger.debug("Frankfurter unavailable, using last known rate for %s/%s", source, target) 

70 return fallback 

71 

72 

73# --------------------------------------------------------------------------- 

74# Public API 

75# --------------------------------------------------------------------------- 

76 

77 

78def get_rate( 

79 con, 

80 rate_date: date, 

81 source: str, 

82 target: str, 

83 *, 

84 offline: bool = False, 

85) -> Decimal: 

86 """Get exchange rate: ``amount_source * rate = amount_target``. 

87 

88 When *offline* is True the DB is checked first; if a rate is already 

89 stored the function returns immediately without any HTTP calls. 

90 This keeps API request latency low — the background prefetch task 

91 populates the DB ahead of time. If the DB has no rate, the normal 

92 online resolution path runs as a fallback. 

93 

94 Uses NBS when one of the currencies is RSD, Frankfurter (ECB) otherwise. 

95 Rates are stored in the exchange_rates table for subsequent lookups. 

96 """ 

97 if source.upper() == target.upper(): 

98 return Decimal(1) 

99 

100 if offline: 

101 db_rate = _get_db_rate(con, rate_date, source, target) 

102 if db_rate is not None: 

103 return db_rate 

104 

105 rsd_involved = source.upper() == "RSD" or target.upper() == "RSD" 

106 

107 if rsd_involved: 

108 rate = _resolve_from_nbs(con, rate_date, source, target) 

109 if rate is not None: 

110 return rate 

111 

112 rate = _resolve_from_frankfurter(con, rate_date, source, target) 

113 if rate is not None: 

114 return rate 

115 

116 msg = f"Could not find rate for {source}/{target} on {rate_date}" 

117 raise ValueError(msg)
