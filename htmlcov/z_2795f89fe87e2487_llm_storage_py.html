164def _providers_from_toml(path: Path) -> list[dict]: 

165 if not path.exists(): 

166 return [] 

167 with path.open("rb") as fh: 

168 data = tomllib.load(fh) 

169 providers = [] 

170 for entry in data.get("providers", []): 

171 base_url = entry.get("base_url", "") 

172 api_key = entry.get("api_key", "") 

173 if not base_url or not api_key: 

174 continue 

175 providers.append( 

176 { 

177 "label": entry.get("label") or _label_from_base_url(base_url), 

178 "base_url": base_url, 

179 "api_key": api_key, 

180 "model": entry.get("model", ""), 

181 "rate_limit_sec": int(entry.get("rate_limit_sec", 60)), 

182 }, 

183 ) 

184 return providers

245def _label_from_base_url(base_url: str) -> str: 

246 lower = base_url.lower() 

247 if "groq" in lower: 

248 return "Groq" 

249 if "openrouter" in lower: 

250 return "OpenRouter" 

251 if "gemini" in lower or "googleapis" in lower or "aistudio" in lower: 

252 return "Gemini" 

253 if "cerebras" in lower: 

254 return "Cerebras" 

255 try: 

256 host = urlparse(base_url).hostname or base_url 

257 return host.split(".")[0].capitalize() 

258 except Exception: # noqa: BLE001 

259 return base_url[:40]
