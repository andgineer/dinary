1"""Runtime 3D -> 2D mapping sourced from a curated ``map`` worksheet tab. 

2 

3The tab is a flat, human-editable table; the drain loop uses it to 

4resolve a 2D target ``(sheet_category, sheet_group)`` for each 3D 

5expense ``(category_id, event_id, tag_ids)``. The evaluation model is 

6"first non-``*`` wins per column, independently": rows are scanned 

7top-to-bottom, every matching row contributes its values, and for each 

8of the two output columns we pick the first non-wildcard value we see. 

9 

10Tab layout (columns A..E, one header row, then data rows): 

11 

12 A category canonical category name, or ``*`` for "any" 

13 B event event name, or ``*`` for "any" (includes no event) 

14 C tags comma- or whitespace-separated tag names that must 

15 ALL be present on the expense; ``*`` / empty = no 

16 required tags 

17 D Расходы target sheet_category; ``*`` / empty means 

18 "don't decide here"; any other value wins 

19 E Конверт same three semantics for sheet_group 

20 

21Fallbacks applied only when no row decides a column: 

22 sheet_category -> ``categories.name`` 

23 sheet_group -> ``""`` 

24 

25The DB tables ``sheet_mapping`` / ``sheet_mapping_tags`` are derived 

26state; the tab is the source of truth. ``reload_now`` swaps them 

27atomically. 

28"""

103def _case_insensitive_match_hint(missing: str, candidates: Iterable[str]) -> str: 

104 lower = missing.casefold() 

105 for candidate in candidates: 

106 if candidate.casefold() == lower: 

107 return f"; did you mean {candidate!r}?" 

108 return ""

111def _parse_tags_cell(raw: str) -> list[str]: 

112 stripped = raw.strip() 

113 if not stripped or stripped == WILDCARD: 

114 return [] 

115 return [part for part in _TAG_SEPARATOR_RE.split(stripped) if part and part != WILDCARD]

119 stripped = raw.strip()

121 return WILDCARD

125def parse_rows( # noqa: C901, PLR0912 

126 raw_rows: list[list[str]], 

127 *, 

128 cat_id_by_name: dict[str, int], 

129 event_id_by_name: dict[str, int], 

130 tag_id_by_name: dict[str, int], 

131) -> list[MapRow]: 

132 """Validate and resolve raw ``map`` tab rows into ``MapRow`` objects. 

133 

134 Skips visually-blank rows so operators can use empty rows as 

135 separators. Resolves category / event / tags against the catalog; 

136 ``*`` / empty cells become wildcards. Raises ``MapTabError`` on 

137 unknown names. 

138 """ 

139 parsed: list[MapRow] = [] 

140 # ``raw_rows`` is ``ws.get_all_values()[1:]`` (header stripped by 

141 # the caller), so ``raw_rows[0]`` is sheet row 2. ``sheet_row`` 

142 # numbers errors the way the operator sees them in Google Sheets. 

143 for sheet_row, row in enumerate(raw_rows, start=2): 

144 cells = list(row) + [""] * max(0, 5 - len(row)) 

145 category_cell = cells[0].strip() 

146 event_cell = cells[1].strip() 

147 tags_cell = cells[2] 

148 sheet_category_raw = cells[3] 

149 sheet_group_raw = cells[4] 

150 

151 if not any( 

152 [ 

153 category_cell, 

154 event_cell, 

155 tags_cell.strip(), 

156 sheet_category_raw.strip(), 

157 sheet_group_raw.strip(), 

158 ], 

159 ): 

160 continue 

161 

162 category_norm = _normalize_cell(category_cell) 

163 event_norm = _normalize_cell(event_cell) 

164 

165 category_id: int | None = None 

166 if category_norm != WILDCARD: 

167 if category_norm not in cat_id_by_name: 

168 hint = _case_insensitive_match_hint( 

169 category_norm, 

170 cat_id_by_name.keys(), 

171 ) 

172 msg = ( 

173 f"map tab row {sheet_row}: category {category_norm!r} is not a known " 

174 f"categories.name (case-sensitive){hint}" 

175 ) 

176 raise MapTabError(msg) 

177 category_id = cat_id_by_name[category_norm] 

178 

179 event_id: int | None = None 

180 if event_norm != WILDCARD: 

181 if event_norm not in event_id_by_name: 

182 hint = _case_insensitive_match_hint( 

183 event_norm, 

184 event_id_by_name.keys(), 

185 ) 

186 msg = ( 

187 f"map tab row {sheet_row}: event {event_norm!r} is not a known " 

188 f"events.name (case-sensitive){hint}" 

189 ) 

190 raise MapTabError(msg) 

191 event_id = event_id_by_name[event_norm] 

192 

193 tag_names = _parse_tags_cell(tags_cell) 

194 tag_ids: list[int] = [] 

195 for tag_name in tag_names: 

196 if tag_name not in tag_id_by_name: 

197 hint = _case_insensitive_match_hint(tag_name, tag_id_by_name.keys()) 

198 msg = ( 

199 f"map tab row {sheet_row}: tag {tag_name!r} is not a known " 

200 f"tags.name (case-sensitive){hint}" 

201 ) 

202 raise MapTabError(msg) 

203 tag_ids.append(tag_id_by_name[tag_name]) 

204 

205 # D / E follow the same wildcard-vs-literal convention as A / B: 

206 # either ``*`` or an empty cell means "don't decide here, defer 

207 # to a later row". An all-literal-explicit-clear case ("keep 

208 # the column blank") is not expressible in the sheet; the 

209 # default fallback (empty string for ``sheet_group``) is what 

210 # operators get when no row assigns a value. 

211 sheet_category = _normalize_cell(sheet_category_raw) 

212 sheet_group = _normalize_cell(sheet_group_raw) 

213 

214 # All-wildcard rows contribute nothing to either output column 

215 # and just burn a ``row_order``. Skip them so reload diagnostics 

216 # and the row count reflect meaningful rules only. 

217 if ( 

218 category_id is None 

219 and event_id is None 

220 and not tag_ids 

221 and sheet_category == WILDCARD 

222 and sheet_group == WILDCARD 

223 ): 

224 continue 

225 

226 parsed.append( 

227 MapRow( 

228 row_order=len(parsed) + 1, 

229 category_id=category_id, 

230 event_id=event_id, 

231 tag_ids=tuple(sorted(set(tag_ids))), 

232 sheet_category=sheet_category, 

233 sheet_group=sheet_group, 

234 ), 

235 ) 

236 return parsed

244def resolve_projection( 

245 rows: Iterable[MapRow], 

246 *, 

247 category_id: int, 

248 event_id: int | None, 

249 tag_ids: set[int], 

250 default_sheet_category: str, 

251) -> tuple[str, str]: 

252 """Apply the "first non-* wins per column" resolver. 

253 

254 ``rows`` must be ordered by ``row_order`` ascending. Returns the 

255 resolved ``(sheet_category, sheet_group)`` pair; an empty string 

256 is a legitimate per-column result (explicit clear). 

257 

258 NOTE: ``ledger_repo.logging_projection`` contains a second copy of 

259 this same semantics (running directly against ``sheet_mapping`` / 

260 ``sheet_mapping_tags`` rows fetched from the DB). The two live 

261 apart so this pure helper can stay as a dependency-free building 

262 block that tests pin, while the DB-backed variant avoids 

263 materializing every ``MapRow`` in memory on every drain. Any 

264 change to the matching rule (new wildcard semantics, reorder 

265 precedence, extra dimensions) **must** be applied in both places. 

266 """ 

267 resolved_category: str | None = None 

268 resolved_group: str | None = None 

269 

270 for row in rows: 

271 if row.category_id is not None and row.category_id != category_id: 

272 continue 

273 if row.event_id is not None and row.event_id != event_id: 

274 continue 

275 if row.tag_ids and not set(row.tag_ids).issubset(tag_ids): 

276 continue 

277 

278 if resolved_category is None and row.sheet_category != WILDCARD: 

279 resolved_category = row.sheet_category 

280 if resolved_group is None and row.sheet_group != WILDCARD: 

281 resolved_group = row.sheet_group 

282 

283 if resolved_category is not None and resolved_group is not None: 

284 break 

285 

286 if resolved_category is None: 

287 resolved_category = default_sheet_category 

288 if resolved_group is None: 

289 resolved_group = "" 

290 return (resolved_category, resolved_group)

298def _load_catalog( 

299 con: sqlite3.Connection, 

300) -> tuple[dict[str, int], dict[str, int], dict[str, int]]: 

301 """Load every catalog row (active or inactive) by name. 

302 

303 The map tab references names that must stay resolvable after an 

304 operator hides a row from the manual PWA picker. ``is_active`` on 

305 a category / event / tag is purely a "hide from the ручной пикер" 

306 affordance; it must not break the mapping reload. Hard-deleted 

307 rows are the only invariant — their names are simply absent and 

308 ``parse_rows`` will still raise ``MapTabError`` with the 

309 "did you mean" hint. 

310 """ 

311 cat_rows = con.execute("SELECT name, id FROM categories").fetchall() 

312 event_rows = con.execute("SELECT name, id FROM events").fetchall() 

313 tag_rows = con.execute("SELECT name, id FROM tags").fetchall() 

314 return ( 

315 {str(r[0]): int(r[1]) for r in cat_rows}, 

316 {str(r[0]): int(r[1]) for r in event_rows}, 

317 {str(r[0]): int(r[1]) for r in tag_rows}, 

318 )

321def _atomic_swap(con: sqlite3.Connection, rows: list[MapRow]) -> None: 

322 """Wipe ``sheet_mapping(_tags)`` and repopulate in a single transaction.""" 

323 con.execute("BEGIN IMMEDIATE") 

324 try: 

325 con.execute("DELETE FROM sheet_mapping_tags") 

326 con.execute("DELETE FROM sheet_mapping") 

327 for row in rows: 

328 con.execute( 

329 "INSERT INTO sheet_mapping" 

330 " (row_order, category_id, event_id, sheet_category, sheet_group)" 

331 " VALUES (?, ?, ?, ?, ?)", 

332 [ 

333 row.row_order, 

334 row.category_id, 

335 row.event_id, 

336 row.sheet_category, 

337 row.sheet_group, 

338 ], 

339 ) 

340 for tag_id in row.tag_ids: 

341 con.execute( 

342 "INSERT INTO sheet_mapping_tags (mapping_row_order, tag_id) VALUES (?, ?)", 

343 [row.row_order, tag_id], 

344 ) 

345 con.execute("COMMIT") 

346 except Exception: 

347 ledger_repo.best_effort_rollback(con, context="sheet_mapping._atomic_swap") 

348 raise

356def reload_now(*, check_after: bool = True) -> dict: 

357 """Unconditional reload: fetch the map tab, parse, swap the DB. 

358 

359 Returns a summary dict with the new ``modifiedTime``, number of 

360 parsed rows, and the tab name. Raises ``MapTabError`` on 

361 parse/validation failure without touching the DB; the current 

362 ``sheet_mapping`` contents stay in place. 

363 

364 ``check_after`` controls the lost-update guard (two Drive metadata 

365 GETs). The drain-loop path uses ``check_after=True`` so a 

366 concurrent edit that shifts ``modifiedTime`` mid-reload leaves 

367 the cache unset and the next tick retries; the admin path uses 

368 ``check_after=False`` to halve the Drive quota cost. 

369 """ 

370 # Normalise the env value — operators may paste the full browser 

371 # URL instead of the bare id. The Sheets + Drive clients want the 

372 # bare id, so extract it here before any API call. 

373 spreadsheet_id = spreadsheet_id_from_setting(settings.sheet_logging_spreadsheet) 

374 if not spreadsheet_id: 

375 msg = "sheet_logging_spreadsheet not configured; nothing to reload" 

376 raise MapTabError(msg) 

377 

378 modified_time_before = drive_get_modified_time(spreadsheet_id) 

379 

380 sh = get_sheet(spreadsheet_id) 

381 try: 

382 ws = sh.worksheet(settings.sheet_mapping_tab_name) 

383 except gspread.WorksheetNotFound as exc: 

384 msg = ( 

385 f"map tab {settings.sheet_mapping_tab_name!r} not found on " 

386 f"spreadsheet {spreadsheet_id!r}; create it via " 

387 "ensure_default_map_tab() or copy the template from docs/" 

388 ) 

389 raise MapTabError(msg) from exc 

390 

391 raw = ws.get_all_values()[1:] 

392 

393 con = ledger_repo.get_connection() 

394 try: 

395 cat_id_by_name, event_id_by_name, tag_id_by_name = _load_catalog(con) 

396 rows = parse_rows( 

397 raw, 

398 cat_id_by_name=cat_id_by_name, 

399 event_id_by_name=event_id_by_name, 

400 tag_id_by_name=tag_id_by_name, 

401 ) 

402 _atomic_swap(con, rows) 

403 finally: 

404 con.close() 

405 

406 global _last_seen_modified_time # noqa: PLW0603 

407 if not check_after: 

408 _last_seen_modified_time = modified_time_before 

409 modified_time_after = modified_time_before 

410 cached = True 

411 else: 

412 modified_time_after = drive_get_modified_time(spreadsheet_id) 

413 if modified_time_after == modified_time_before: 

414 _last_seen_modified_time = modified_time_after 

415 cached = True 

416 else: 

417 logger.info( 

418 "sheet_mapping: modifiedTime shifted during reload (%s -> %s); " 

419 "leaving cache unset so the next ensure_fresh() retries", 

420 modified_time_before, 

421 modified_time_after, 

422 ) 

423 cached = False 

424 

425 summary: dict = { 

426 "row_count": len(rows), 

427 "modified_time": modified_time_after, 

428 "tab": settings.sheet_mapping_tab_name, 

429 "modified_time_cached": cached, 

430 } 

431 logger.info("sheet_mapping reloaded: %s", summary) 

432 return summary

435def ensure_fresh() -> None: 

436 """Drain-loop hook: reload the map tab iff Drive says it changed.""" 

437 spreadsheet_id = spreadsheet_id_from_setting(settings.sheet_logging_spreadsheet) 

438 if not spreadsheet_id: 

439 return 

440 try: 

441 modified_time = drive_get_modified_time(spreadsheet_id) 

442 except Exception: # noqa: BLE001 — see comment below 

443 # Drive failures are routine (token refresh, 5xx, network 

444 # blips, gspread/google-auth transient errors) and we 

445 # deliberately swallow *all* of them here: the cached 

446 # ``sheet_mapping`` keeps serving writes while we wait for 

447 # the next drain tick. ``exc_info=True`` preserves the 

448 # traceback so a persistent failure still has enough 

449 # context for triage. 

450 logger.warning( 

451 "sheet_mapping: drive_get_modified_time failed; keeping cached sheet_mapping", 

452 exc_info=True, 

453 ) 

454 return 

455 # Snapshot the module global once. Without this, a concurrent 

456 # ``reload_now`` between the equality check and the log line below 

457 # would print mismatched "before" values. 

458 last_seen = _last_seen_modified_time 

459 if modified_time == last_seen: 

460 return 

461 logger.info( 

462 "sheet_mapping: map-tab changed (%s -> %s); reloading", 

463 last_seen, 

464 modified_time, 

465 ) 

466 try: 

467 reload_now() 

468 except Exception: 

469 logger.exception( 

470 "sheet_mapping: reload_now() failed; keeping cached mapping", 

471 )

508# Tag-driven envelope rules (category=*, event=*, Расходы=*). Ordered: 

509# beneficiary-specific tags first, then sphere-of-life tags. Any 

510# expense carrying the tag lands in the matching envelope regardless 

511# of category. 

512_TAG_RULES: list[tuple[str, str]] = [ 

513 ("Лариса", "лариса"), 

514 ("Аня", "ребенок"), 

515 ("собака", "собака"), 

516 # Vacation events attach both "отпуск" and "путешествия" auto-tags, 

517 # so either on its own resolves to the "путешествия" envelope. 

518 # Keep both rules so a manual tag-only classification (without 

519 # attaching an event) still routes to the right envelope. 

520 ("отпуск", "путешествия"), 

521 ("путешествия", "путешествия"), 

522 ("релокация", "релокация"), 

523 ("дача", "дача"), 

524 ("профессиональное", "профессиональное"), 

525] 

526 

527 

528# Per-category envelope overrides. Emitted only for categories whose 

529# Конверт is not the resolver's default blank. ``Расходы`` is left as 

530# ``WILDCARD`` because the "no rule matched" fallback in both 

531# ``resolve_projection`` (sheet_mapping.py) and 

532# ``logging_projection`` (ledger_repo.py) already substitutes the 

533# category's canonical name — so a literal ``cname`` here would just 

534# be noise duplicating a fallback that is exercised by tests. 

535_CATEGORY_ENVELOPES: dict[str, str] = { 

536 "гигиена": "гигиена", 

537 "ЗОЖ": "ЗОЖ", 

538}

541def _default_template_rows( 

542 category_names: list[str], 

543 *, 

544 active_tag_names: set[str] | None = None, 

545) -> list[list[str]]: 

546 """Produce the body rows for the default map tab, in evaluation order. 

547 

548 1. Generic tag → envelope rules (category=*, event=*, Расходы=*). 

549 Rules referencing a tag name not in ``active_tag_names`` are 

550 dropped with a WARN — a rename in the catalog without a 

551 corresponding ``_TAG_RULES`` update would otherwise emit a 

552 template that trips ``MapTabError`` on first read. 

553 2. Per-category envelope overrides, *one row per entry in* 

554 ``_CATEGORY_ENVELOPES``: Расходы stays ``*`` (the resolver falls 

555 back to ``category.name``), Конверт takes the override value. 

556 Pure identity rows (Расходы = category, Конверт = ``*``) are 

557 deliberately not emitted — they would be indistinguishable from 

558 the resolver's no-rule-matched fallback and only bloat the tab. 

559 

560 ``active_tag_names`` defaults to "skip filtering" so call sites 

561 that only have category names (tests, older call sites) keep 

562 working; callers that have the active catalog handy pass it in 

563 to get the filtered template. 

564 

565 ``category_names`` is consulted only to filter 

566 ``_CATEGORY_ENVELOPES`` against the active catalog: an override 

567 for a category that is no longer active is dropped with a WARN 

568 rather than emitted (the parser would reject it as an unknown 

569 category name anyway). 

570 """ 

571 rows: list[list[str]] = [] 

572 for tag, envelope in _TAG_RULES: 

573 if active_tag_names is not None and tag not in active_tag_names: 

574 logger.warning( 

575 "ensure_default_map_tab: skipping tag rule %r -> %r " 

576 "because the tag is not in the active catalog; " 

577 "update sheet_mapping._TAG_RULES after renaming tags", 

578 tag, 

579 envelope, 

580 ) 

581 continue 

582 rows.append([WILDCARD, WILDCARD, tag, WILDCARD, envelope]) 

583 active_category_names = set(category_names) 

584 for cname, envelope in _CATEGORY_ENVELOPES.items(): 

585 if cname not in active_category_names: 

586 logger.warning( 

587 "ensure_default_map_tab: skipping envelope override %r -> %r " 

588 "because the category is not in the active catalog; " 

589 "update sheet_mapping._CATEGORY_ENVELOPES after renaming categories", 

590 cname, 

591 envelope, 

592 ) 

593 continue 

594 rows.append([cname, WILDCARD, WILDCARD, WILDCARD, envelope]) 

595 return rows

598def ensure_default_map_tab() -> None: 

599 """Create the ``map`` worksheet tab with a default template if missing. 

600 

601 Idempotent: a second call dry-runs the existing tab against the 

602 current active catalog and WARN-logs any stale references rather 

603 than mutating. 

604 """ 

605 spreadsheet_id = spreadsheet_id_from_setting(settings.sheet_logging_spreadsheet) 

606 if not spreadsheet_id: 

607 logger.info("ensure_default_map_tab: sheet_logging_spreadsheet empty; skipping") 

608 return 

609 sh = get_sheet(spreadsheet_id) 

610 try: 

611 existing_ws = sh.worksheet(settings.sheet_mapping_tab_name) 

612 except gspread.WorksheetNotFound: 

613 pass 

614 else: 

615 con = ledger_repo.get_connection() 

616 try: 

617 _warn_if_existing_map_tab_is_stale(existing_ws, con) 

618 finally: 

619 con.close() 

620 return 

621 

622 con = ledger_repo.get_connection() 

623 try: 

624 cat_rows = con.execute( 

625 "SELECT c.name FROM categories c" 

626 " JOIN category_groups g ON g.id = c.group_id" 

627 " WHERE c.is_active AND g.is_active" 

628 " ORDER BY g.sort_order, c.name", 

629 ).fetchall()

632 ).fetchall()

677def decode_auto_tags_value(raw: object, *, context: str = "") -> list[str]: 

678 """Decode a raw ``events.auto_tags`` JSON value into a list of names. 

679 

680 Single canonical implementation shared by every code path that 

681 reads ``events.auto_tags`` (``api/catalog``, ``catalog_writer``, 

682 ``sheet_mapping.load_event_auto_tag_names``). Blank / NULL / 

683 malformed payloads degrade to ``[]`` so a partially migrated DB 

684 cannot wedge the read path; a WARN is logged once per bad value 

685 with the caller-supplied ``context`` string so operators can trace 

686 which event / endpoint surfaced the issue. 

687 """ 

688 if raw is None or raw == "": 

689 return [] 

690 try: 

691 parsed = json.loads(raw if isinstance(raw, str) else str(raw)) 

692 except json.JSONDecodeError: 

693 logger.warning( 

694 "events.auto_tags%s is not valid JSON (%r); treating as empty", 

695 f" for {context}" if context else "", 

696 raw, 

697 ) 

698 return [] 

699 if not isinstance(parsed, list): 

700 logger.warning( 

701 "events.auto_tags%s is not a JSON list (%r); treating as empty", 

702 f" for {context}" if context else "", 

703 raw, 

704 ) 

705 return [] 

706 return [str(name) for name in parsed]

709def load_event_auto_tag_names( 

710 con: sqlite3.Connection, 

711 event_id: int, 

712) -> list[str]: 

713 """Return the JSON-decoded ``events.auto_tags`` list for ``event_id``. 

714 

715 Empty list when the event has no auto-tags, the column is blank, or 

716 the event does not exist. 

717 """ 

718 row = con.execute( 

719 "SELECT auto_tags FROM events WHERE id = ?", 

720 [event_id], 

721 ).fetchone() 

722 if row is None: 

723 return [] 

724 return decode_auto_tags_value(row[0], context=f"event_id={event_id}")

727def resolve_event_auto_tag_ids( 

728 con: sqlite3.Connection, 

729 event_id: int, 

730) -> list[int]: 

731 """Return catalog ids for the event's ``auto_tags`` names. 

732 

733 The returned ids preserve the authoring order from 

734 ``events.auto_tags`` (dedup by first occurrence). Tag rows are 

735 matched by name regardless of their ``is_active`` flag — 

736 ``tags.is_active = FALSE`` means "hide from the ручной пикер" and 

737 must not break event-driven auto-attach. Only names that are 

738 absent from the ``tags`` table entirely (hard-deleted or typo) 

739 are logged at WARN and dropped; we never block an expense write 

740 because the operator hid a tag from the picker. 

741 """ 

742 names = load_event_auto_tag_names(con, event_id) 

743 if not names: 

744 return [] 

745 ordered_unique: list[str] = list(dict.fromkeys(names)) 

746 placeholders = ",".join(["?"] * len(ordered_unique)) 

747 rows = con.execute( 

748 f"SELECT id, name FROM tags WHERE name IN ({placeholders})", # noqa: S608 

749 ordered_unique, 

750 ).fetchall() 

751 found = {str(r[1]): int(r[0]) for r in rows} 

752 missing = [n for n in ordered_unique if n not in found] 

753 if missing: 

754 logger.warning( 

755 "events.auto_tags for event_id=%d references unknown tag names %r; dropping", 

756 event_id, 

757 missing, 

758 ) 

759 return [found[name] for name in ordered_unique if name in found]
